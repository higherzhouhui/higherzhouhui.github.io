function validPhone(phone: string) {
  var reg='^((13[0-9])|(14[5,7])|(15[0-3,5-9])|(17[0,3,5-9])|(18[0-9])|166|198|199|191|(147))\\d{8}$';
  var regExp=new RegExp(reg);
  return regExp.test(phone);
},
